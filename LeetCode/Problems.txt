 Find the Index of the First Occurrence in a String

============
def strStr(haystack: str, needle: str) -> int:
    len_h, len_n = len(haystack), len(needle)
    
    if len_n == 0:
        return 0

    for i in range(len_h - len_n + 1):
        if haystack[i:i+len_n] == needle:
            return i
    return -1
